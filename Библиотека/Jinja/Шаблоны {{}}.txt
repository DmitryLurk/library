from jinja2 import Template

Класс Template используется для формирования шаблона. Он делает подстановку внутрь {{}} переданных аргументов при render

name = olesha

tm = Template("Привет {{ name }}")

msg = tm.render(name=name)	При передаче именованных аргументов в метод render подстановка выполниться автоматически

print(msg)

{%%} спецификатор шаблона
{{}} выражение для вставки кострукций Python в шаблон
{##} блок комментариев
# ## строковый комментарий
